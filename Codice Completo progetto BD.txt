CREATE OR REPLACE FUNCTION checkCamp()
RETURNS TRIGGER AS $$
DECLARE 
    CampAnalizzato int;
BEGIN
    SELECT COUNT(*) INTO CampAnalizzato
    FROM Competizione
    WHERE nomeCompetizione = NEW.campAppartenenza 
    AND Competizione.nazionalita = NEW.nazionalita;

    IF CampAnalizzato = 0 THEN 
        RAISE EXCEPTION 'Un eccezione è stata violata: Il campionato !';
    END IF;

    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER checkSquadraCamp 
BEFORE INSERT ON SQUADRA 
FOR EACH ROW 
EXECUTE FUNCTION checkCamp();

CREATE OR REPLACE FUNCTION SearchPosition()
RETURNS TRIGGER AS $$
DECLARE
    CheckPosNEXT int;
BEGIN
    SELECT COUNT(*)
    INTO CheckPosNEXT
    FROM Partecipa
    WHERE nomeCompetizione = NEW.nomeCompetizione
      AND posizioneFinale = NEW.posizioneFinale
      AND nomeSquadra <> NEW.nomeSquadra;

    IF CheckPosNEXT > 0 THEN
        RAISE EXCEPTION 'Un eccezione è stata violata: due squadre non possono avere la stessa posizione finale!';
    END IF;

    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER CheckPosition
BEFORE INSERT ON Partecipa
FOR EACH ROW
EXECUTE FUNCTION SearchPosition();

-- Creare la sequenza
CREATE SEQUENCE trofeo_di_squadra_seq START 1;

-- Creare la funzione per generare e formattare l'ID
CREATE OR REPLACE FUNCTION genera_id_trofeo()
RETURNS VARCHAR(8) AS $$
DECLARE
    nuovo_id INT;
    id_formattato VARCHAR(8);
BEGIN
    -- Ottenere il prossimo valore dalla sequenza
    nuovo_id = nextval('trofeo_di_squadra_seq');
    
    -- Formattare il valore come 'DS' seguito da un numero a 6 cifre
    id_formattato = 'DS' || lpad(nuovo_id::text, 6, '0');
    
    RETURN id_formattato;
END;
$$ LANGUAGE plpgsql;

-- Creare la funzione trigger

    IF NEW.posizioneFinale = 1 THEN

    END IF;
    
    RETURN NEW;
END;

CREATE OR REPLACE FUNCTION assegna_premio_calciatori()
RETURNS TRIGGER AS $$
DECLARE
    DataFinale DATE;
    ID_Calciatore INTEGER;

    scansionaGiocatore CURSOR FOR
        SELECT ID
        FROM Carriera
        WHERE nomeSquadra = NEW.nomeSquadra
          AND CAST(EXTRACT(YEAR FROM dataInizioCarriera) AS INTEGER)
 <= CAST(SPLIT_PART(NEW.annoSvolgimento, '/', 1) AS INTEGER);
BEGIN
    -- Seleziona la data massima della partita
    SELECT MAX(data) INTO DataFinale
    FROM PARTITA
    WHERE nomeSquadra = NEW.nomeSquadra
      AND annoSvolgimento = NEW.annoSvolgimento
      AND nomeCompetizione = NEW.nomeCompetizione;
    
    -- Apri il cursore
    OPEN scansionaGiocatore;

    -- Ciclo per ogni calciatore nel cursore
    LOOP
        FETCH scansionaGiocatore INTO ID_Calciatore;
        EXIT WHEN NOT FOUND;
        -- Inserisce il premio nella tabella VINCE
        INSERT INTO VINCE (ID_TROFEO_IN, ID_TROFEO_DS, ID, dataVincita)
        VALUES (NULL, NEW.ID_TROFEO_DS, ID_Calciatore, DataFinale);
    END LOOP;

    -- Chiudi il cursore
    CLOSE scansionaGiocatore;

    RETURN NEW;
END;
$$ LANGUAGE plpgsql;
-- Creazione della trigger associata alla tabella TROFEO_DI_SQUADRA
CREATE TRIGGER assegna_premio
AFTER INSERT ON TROFEO_DI_SQUADRA
FOR EACH ROW
EXECUTE FUNCTION assegna_premio_calciatori();

RETURNS TRIGGER AS $$
DECLARE
    CheckTotalID int;
BEGIN
    -- Conta il numero di squadre con lo stesso ID_partita
    SELECT COUNT(*) INTO CheckTotalID
    FROM Partita
    WHERE ID_Partita = NEW.ID_partita;

    -- non possono esserci più di due squadre nella stessa partita
    IF CheckTotalID >= 2 THEN
        RAISE EXCEPTION 'Solo due squadre possono partecipare alla stessa partita!';
    END IF;

    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE OR REPLACE FUNCTION Check_inCasaCorrect()
RETURNS TRIGGER AS $$
DECLARE
    CheckTotal_inCasa int;
    StadioSquadra varchar(100);
BEGIN

    
    IF NEW.inCasa = TRUE THEN
      
        SELECT COUNT(*) INTO CheckTotal_inCasa
        FROM Partita
        WHERE inCasa = TRUE
        AND ID_partita = NEW.ID_Partita;

        
        IF CheckTotal_inCasa > 0 THEN
            RAISE EXCEPTION 'Solo una squadra può giocare in casa per la stessa partita!';
        END IF;

        
        SELECT S.nomeStadio INTO StadioSquadra
        FROM SQUADRA AS S
        WHERE S.nomeSquadra = NEW.nomeSquadra;

       
        IF NEW.nomeStadio <> StadioSquadra THEN
            RAISE EXCEPTION 'Lo stadio della partita non è associato alla squadra che gioca in casa';
        END IF;
    END IF;

    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

-- Definizione del trigger
CREATE TRIGGER Check_inCasa
BEFORE INSERT ON Partita
FOR EACH ROW
EXECUTE FUNCTION Check_inCasaCorrect();

CREATE OR REPLACE FUNCTION ExistRoleAndPlay()
RETURNS TRIGGER AS $$
DECLARE 
    PosizioneDaControllare VARCHAR(30);
	RuoloConfermato VARCHAR(10);
BEGIN
	IF NEW.posizione <> 'Allenatore' THEN
		IF NOT EXISTS (SELECT 1 FROM RUOLO WHERE nomeRuolo = NEW.posizione) THEN
			RAISE EXCEPTION 'Il ruolo inserito non esiste!';
		END IF;

		
		SELECT G.abbrRuolo INTO RuoloConfermato
		FROM GIOCA AS G
		JOIN RUOLO AS R ON G.abbrRuolo = R.abbrRuolo
		WHERE G.ID = NEW.ID_Calciatore AND R.nomeRuolo = NEW.posizione;

		IF RuoloConfermato IS NULL THEN
			RAISE NOTICE 'Il giocatore non gioca quel ruolo! posizione: %, ID_Calciatore: %', NEW.posizione, NEW.ID_Calciatore;

		END IF;
	END IF;
	RETURN NEW;
	
END;
$$ LANGUAGE plpgsql;

CREATE OR REPLACE FUNCTION UpdatePlayerCareer()
RETURNS TRIGGER AS $$
DECLARE 
    isGoalKeeper BOOLEAN;
    goal_segnati INTEGER;
	goal_subiti_dall_avversaria INTEGER;
    ResultMatch INTEGER;
BEGIN
    IF NEW.cartellinoRosso IS TRUE THEN
        UPDATE CARRIERA
        SET cartelliniRossiAnnuali = cartelliniRossiAnnuali + 1
        WHERE NEW.ID = CARRIERA.ID;
    END IF;

    IF NEW.cartelliniGialli > 0 THEN
        UPDATE CARRIERA
        SET cartelliniGialliAnnuali = cartelliniGialliAnnuali + NEW.cartelliniGialli
        WHERE NEW.ID = CARRIERA.ID;
    END IF;

    IF NEW.infortunato IS TRUE THEN
        UPDATE CARRIERA
        SET infortuniAnnuali = infortuniAnnuali + 1
        WHERE NEW.ID = CARRIERA.ID;
    END IF;

    IF NEW.goalEseguiti > 0 THEN
        UPDATE CARRIERA
        SET goalEseguitiAnnuali = goalEseguitiAnnuali + NEW.goalEseguiti
        WHERE NEW.ID = CARRIERA.ID;
    END IF;

    SELECT EXISTS (
        SELECT 1
        FROM GIOCA
        WHERE NEW.ID = GIOCA.ID
        AND abbrRuolo = 'POR'
    ) INTO isGoalKeeper;

    IF isGoalKeeper THEN
    SELECT 
    SUM(c1.goalEseguiti),
    SUM(c2.goalSubiti) INTO goal_segnati, goal_subiti_dall_avversaria
	FROM PARTITA p1
	JOIN COMPETE c1 ON p1.ID_Partita = c1.ID_Partita
	JOIN PARTITA p2 ON p1.ID_Partita = p2.ID_Partita AND p1.nomeSquadra <> p2.nomeSquadra
	JOIN COMPETE c2 ON p2.ID_Partita = c2.ID_Partita
	WHERE p1.inCasa = TRUE;

    IF goal_segnati < goal_subiti_dall_avversaria THEN
		RAISE EXCEPTION 'Il totale di goal subiti non è corretto con il risultato ottenuto dalla partita!';
	END IF;	
	
        UPDATE CARRIERA
        SET goalSubitiAnnuali = goalSubitiAnnuali + NEW.goalSubiti
        WHERE NEW.ID = CARRIERA.ID;
    END IF;

    RETURN NULL;
END;
$$ LANGUAGE plpgsql;

CREATE OR REPLACE FUNCTION CheckRitiroPlayerCorrect()
RETURNS TRIGGER AS $$
DECLARE
    calciatore_ritirato_exists BOOLEAN;
BEGIN
    -- Verifica se il giocatore non è un 'Calciatore' e controlla che esista un ritiro da 'Calciatore'
    IF (NEW.tipologia <> 'Calciatore') THEN
        -- Controlla se esiste un record in CARRIERA per il giocatore con 'tipologia' 'Calciatore' e 'dataRitiro' non nulla
        SELECT EXISTS (
            SELECT 1
            FROM CARRIERA
            WHERE ID = NEW.ID
              AND tipologia = 'Calciatore'
              AND dataRitiro IS NOT NULL
        ) INTO calciatore_ritirato_exists;

        -- Se non esiste, solleva un'eccezione
        IF NOT calciatore_ritirato_exists THEN
            RAISE EXCEPTION 'Il giocatore non risulta Ritirato dal ruolo Calciatore.';
        END IF;
    END IF;

    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

BEFORE INSERT ON CARRIERA
FOR EACH ROW 
EXECUTE FUNCTION CheckRitiroPlayerCorrect();
